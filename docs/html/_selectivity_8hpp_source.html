44  template<typename REAL_T>
45  struct SelectivityBase : mas::ModelObject<REAL_T> {
46  typedef typename VariableTrait<REAL_T>::variable variable;
47  std::vector<REAL_T> lambdas;
48  variable sigma = 0.85;
49  variable sigma2 = 0.7225;
50  std::vector<variable> selectivity;
51 
52  void Update(const std::vector<variable>& ages) {
53  if (this->selectivity.size() != ages.size()) {
54  this->selectivity.resize(ages.size());
55  }
56  for (size_t i = 0; i < this->selectivity.size(); i++) {
57  this->selectivity[i] = this->Evaluate(ages, i);
58  }
59  }

63  }

70  return "SelectivityBase";
71  }

73  virtual variable LikelihoodComponent(int phase) {
74 
75  assert(this->initial_parameter_values.size() == this->estimated_parameters.size());
76  variable ret;
77  for (int i = 0; i < this->estimated_parameters.size(); i++) {
78  if (this->estimated_phase[i] <= phase) {
79  variable p = *this->estimated_parameters[i];
80  if (p > 0.0) {
81  ret += (atl::log(sigma) + 0.5 * SELX_SQUARE(std::log(this->initial_parameter_values[i]) - atl::log(p)) / sigma2);
82  } else {
83  std::cout << "Warning: cannot do prior in log space for parm with min <= 0.0\n";
84  }
85  }
86  }
87 
88  return ret;
89  }

94  struct LogisticSel : SelectivityBase<REAL_T> {
95  typedef typename VariableTrait<REAL_T>::variable variable;
96  variable a50; //age of 50% selectivity
97  variable s; //the rate of increase in selectivity at a50
98 
104  virtual const variable Evaluate(const variable& a) {
105  return static_cast<REAL_T> (1.0) / (static_cast<REAL_T> (1.0) + atl::exp(-1.0 * (a - a50) / s));
106  }
107 
108  virtual const variable Evaluate(const std::vector<variable>& ages, size_t index) {
109  return static_cast<REAL_T> (1.0) / (static_cast<REAL_T> (1.0) + atl::exp(-1.0 * (ages[index] - a50) / s));
110  }

138  struct DoubleLogisticSel : SelectivityBase<REAL_T> {
139  typedef typename VariableTrait<REAL_T>::variable variable;
140  variable alpha_asc; //ascending alpha
141  variable beta_asc; // ascending beta
142  variable alpha_desc; // descending alpha
143  variable beta_desc; // descending beta
144 
150  virtual const variable Evaluate(const variable& a) {
151  return (static_cast<REAL_T> (1.0) /
152  (static_cast<REAL_T> (1.0) +
153  atl::mfexp(-beta_asc * (a - alpha_asc)))) *
154  (static_cast<REAL_T> (1.0) - (static_cast<REAL_T> (1.0) /
155  (static_cast<REAL_T> (1.0) +
156  atl::mfexp(-beta_desc * (a - alpha_desc)))));
157  }
158 
159  virtual const variable Evaluate(const std::vector<variable>& ages, size_t index) {
160  return (static_cast<REAL_T> (1.0) /
161  (static_cast<REAL_T> (1.0) +
162  atl::mfexp(-beta_asc * (ages[index] - alpha_asc)))) *
163  (static_cast<REAL_T> (1.0) - (static_cast<REAL_T> (1.0) /
164  (static_cast<REAL_T> (1.0) +
165  atl::mfexp(-beta_desc * (ages[index] - alpha_desc)))));

254  return atl::pow((centroid - a)*(centroid - a), 0.5);
255  }
256 
262  virtual const variable Evaluate(const variable& a) {
263  return atl::mfexp(-1.0 * atl::pow(this->epsilon * Distance(a), 2.0));
