178  const variable dnorm(const variable& x,
179  const variable& mean,
180  const variable& sd, int give_log = 0) {
181  variable SD = sd;
182  if (sd.GetValue() == 0.0) {
183  SD += 1e-8;
184  throw std::overflow_error("Divide by zero exception");
185  }
186 
187  variable logres = static_cast<double> (-1.0) * atl::log(double(std::sqrt(2 * M_PI)) * SD) - static_cast<double> (.5) * atl::pow((x - mean) / SD, static_cast<double> (2.0));
188  if (give_log)return logres;
189  else return atl::exp(logres);
190  }

192  static REAL_T PolynomialInterpolator(REAL_T x,
193  const std::vector<REAL_T>& xi,
194  const std::vector<REAL_T>& yi,
195  int imax) {
196 
197  REAL_T y;
198  int j;
199  // if x is ouside the xi[] interval
200  if (x <= xi[0]) return y = yi[0];
201  if (x >= xi[imax - 1]) return y = yi[imax - 1];
202  // loop to find j so that x[j-1] < x < x[j]
203  j = 0;
204  while (j <= imax - 1) {
205  if (xi[j] >= x) break;
206  j = j + 1;
207  }
208  y = yi[j - 1]+(yi[j] - yi[j - 1])*(x - xi[j - 1]) / (xi[j] - xi[j - 1]);
209  return y;
210  }

212  static void Do2DInterpolation(int id, const std::shared_ptr<mas::DataObject<REAL_T> > & data,
213  std::unordered_map<int, std::unordered_map<int, std::unordered_map<REAL_T, REAL_T> > >& results) {
214 
215  size_t years = data->imax;
216  size_t ages = data->kmax;
217  std::vector<REAL_T> x(ages);
218  REAL_T to_kg = data->Convert(mas::KG);
219  for (size_t y = 0; y < years; y++) {
220  for (size_t a = 0; a < ages; a++) {
221  x[a] = data->get(y, a);
222  results[id][y][mas::GrowthBase<REAL_T>::ages[a]] = x[a] * to_kg;
223  }
224 
225  for (ages_iterator ait = mas::GrowthBase<REAL_T>::ages_to_intrpolate.begin();
226  ait != mas::GrowthBase<REAL_T>::ages_to_intrpolate.end();
227  ++ait) {
228  REAL_T xi = mas::GrowthBase<REAL_T>::PolynomialInterpolator((*ait), mas::GrowthBase<REAL_T>::ages, x, x.size());
229  results[id][y][(*ait)] = xi;
230  }
231  }
