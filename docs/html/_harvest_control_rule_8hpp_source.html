74  // find the fishing mortality F such that the ratio of spawning biomass at F to the unfished
75  // equilibrium spawning biomass is spr_fraction
76  inline std::pair<variable, variable> CalculateFatSPRfraction(const variable& spr_fraction, const std::vector<variable>& fsh_sel, const mas::Subpopulation<REAL_T>& api)
77  {
78  variable Fval = 0.0;
79  variable SBval = 0.0;
80 
81  variable df = static_cast<REAL_T> (1.0e-3);
82  variable Flow, Fhigh, Yrate, Ylow, Yhigh, deltaY, deltaYprime;
83 
84  variable SB0 = CalculateSBatF(0.0, fsh_sel, api);
85 
86  Fval = 0.2; // initial value
87 
88  // Newton-Raphson
89  for (int i = 0; i < 10 && Fval > 0.0; ++i)
90  {
91  Fhigh = Fval + df;
92  Flow = Fval - df;
93 
94  Yrate = static_cast<REAL_T> (-1000.0) * atl::pow(atl::log( spr_fraction / (CalculateSBatF(Fval, fsh_sel, api) / SB0) ), 2.0);
95  Yhigh = static_cast<REAL_T> (-1000.0) * atl::pow(atl::log( spr_fraction / (CalculateSBatF(Fhigh, fsh_sel, api) / SB0) ), 2.0);
96  Ylow = static_cast<REAL_T> (-1000.0) * atl::pow(atl::log( spr_fraction / (CalculateSBatF(Flow, fsh_sel, api) / SB0) ), 2.0);
97 
98  deltaY = (Yhigh - Ylow) / (static_cast<REAL_T> (2.0) * df); // First derivative (to find the root of this)
99  deltaYprime = (Yhigh + Ylow - (static_cast<REAL_T> (2.0) * Yrate)) / (df * df); // Newton-Raphson approximation 2nd deriv
100  Fval -= (deltaY / deltaYprime);
101  }
102 
103  if (Fval < 0.0)
104  {
105  std::cout << "Calculation Warning: F is less than 0 in HCR::CalculateFatSPRfraction for SPR fraction " << spr_fraction << "; setting F to 0\n";
106  mas::mas_log << "Calculation Warning: F is less than 0 in HCR::CalculateFatSPRfractionfor SPR fraction " << spr_fraction << "; setting F to 0\n";
107 
108  Fval = 0.0;
109  }
110 
111  SBval = CalculateSBatF(Fval, fsh_sel, api);
112 
113  return std::make_pair(Fval, SBval);
114  }

116  // calculate the equilibrium female spawning biomass at a specific fishing mortality
117  inline variable CalculateSBatF(const variable& trialF, const std::vector<variable>& fsh_sel, const mas::Subpopulation<REAL_T>& api)
118  {
119  variable SBatF = 0.0;
120 
121  auto Nages = api.ages.size();
122 
123  std::vector<variable> Ztmp(Nages);
124  std::vector<variable> Ntmp(Nages);
125 
126  for (int a = 0; a < Nages; ++a)
127  {
128  Ztmp[a] = (fsh_sel[a] * trialF) + api.M[a];
129  }
130 
131  // R0 could be changed to the average recruitment over a specific period
132  Ntmp[0] = static_cast<REAL_T> (0.5) * api.recruitment_model->R0;
133  SBatF += (Ntmp[0] * api.maturity[0] * api.weight_at_spawning[0] * atl::exp(static_cast<REAL_T> (-1.0) * api.spawning_season_offset * Ztmp[0]));
134 
135  for (int a = 1; a < (Nages-1); ++a)
136  {
137  Ntmp[a] = Ntmp[a-1] * atl::exp(static_cast<REAL_T> (-1.0) * Ztmp[a-1]);
138  SBatF += (Ntmp[a] * api.maturity[a] * api.weight_at_spawning[a] * atl::exp(static_cast<REAL_T> (-1.0) * api.spawning_season_offset * Ztmp[a]));
139  }
140  Ntmp[Nages-1] = Ntmp[Nages-2] * atl::exp(static_cast<REAL_T> (-1.0) * Ztmp[Nages-2]) /
141  (static_cast<REAL_T> (1.0) - atl::exp(static_cast<REAL_T> (-1.0) * Ztmp[Nages-1]));
142  SBatF += (Ntmp[Nages-1] * api.maturity[Nages-1] * api.weight_at_spawning[Nages-1] * atl::exp(static_cast<REAL_T> (-1.0) * api.spawning_season_offset * Ztmp[Nages-1]));
143 
144  return (SBatF / api.kg_to_tonnes);
145  }

147  // this function calculates ABC and OFL for year 'year' using the numbers-at-age at the beginning of the year
148  const std::tuple<variable, variable, variable, variable> Evaluate(const int& year, const mas::Subpopulation<REAL_T>& api_males, const mas::Subpopulation<REAL_T>& api_females)
149  {
150  const REAL_T Tier3_alpha = static_cast<REAL_T> (0.05);
151 
152  const int Nages = api_females.ages.size();
153 
154  // initialize
155  F_ABC = 0.0;
156  ABC = 0.0;
157  F_OFL = 0.0;
158  OFL = 0.0;
159 
160  // NOTE: season is HARD-CODED to 1
161  const int season = 1;
162  std::vector< std::shared_ptr<Fleet<REAL_T> > >& fleets_f = api_females.area->seasonal_fleet_operations[season];
163  std::vector< std::shared_ptr<Fleet<REAL_T> > >& fleets_m = api_males.area->seasonal_fleet_operations[season];
164 
165  if (fleets_f.size() > 0)
166  {
167  std::vector<variable> fsh_sel_f(Nages);
168  std::vector<variable> fsh_sel_m(Nages);
169 
170  variable Zf_ABC, expZf_ABC, Zm_ABC, expZm_ABC, Zf_OFL, expZf_OFL, Zm_OFL, expZm_OFL;
171 
172  variable sp_biomass;
173 
174  // NOTE: fleet number is HARD-CODED to 0
175  const int fleet = 0;
176  for (int a = 0; a < Nages; ++a)
177  {
178  fsh_sel_f[a] = fleets_f[fleet]->season_area_selectivity[season][api_females.area->id]->Evaluate(api_females.ages[a]);
179  fsh_sel_m[a] = fleets_m[fleet]->season_area_selectivity[season][api_males.area->id]->Evaluate(api_males.ages[a]);
180  }
181 
182  // F100 (BRP100.first) had better be close to 0 and
183  // SB100 (BRP100.second) had better be close to api.SB0
184  BRP100 = CalculateFatSPRfraction(1.00, fsh_sel_f, api_females);
185  // std::cout << "SB0 " << api_females.SB0 << std::endl;
186  // std::cout << "SB100 " << BRP100.second << std::endl;
187  // std::cout << "F100 " << BRP100.first << std::endl;
188 
189  BRP40 = CalculateFatSPRfraction(0.40, fsh_sel_f, api_females);
190  BRP35 = CalculateFatSPRfraction(0.35, fsh_sel_f, api_females);
191  BRP20 = CalculateFatSPRfraction(0.20, fsh_sel_f, api_females);
192  BRP05 = CalculateFatSPRfraction(0.05, fsh_sel_f, api_females);
193 
194 
195  // GOA walleye pollock HCR has a different SBtarget and SBfloor
196  // variable SBtarget = BRP40.second * (BRP35.first / BRP40.first);
197  // variable SBfloor = BRP20.second;
198  variable SBtarget = BRP40.second;
199  variable SBfloor = BRP05.second;
200 
201  variable FtargetABC = BRP40.first;
202  variable FtargetOFL = BRP35.first;
203 
204  // initial value
205  F_ABC = FtargetABC;
206  ABC = 0.0;
207  F_OFL = FtargetOFL;
208  OFL = 0.0;
209 
210  // 30 iterations to stabilize the F_ABC and F_OFL values
211  for (int i = 0; i < 30; ++i)
212  {
213  // calculate spawning biomass in year 'year' at F_ABC
214  sp_biomass = 0.0;
215  for (int a = 0; a < Nages; ++a)
216  {
217  Zf_ABC = api_females.M[a] + (F_ABC * fsh_sel_f[a]);
218  expZf_ABC = atl::exp(static_cast<REAL_T> (-1.0) * api_females.spawning_season_offset * Zf_ABC);
219  sp_biomass += (api_females.N_proj[(season - 1) * Nages + a] *
220  api_females.maturity[a] * api_females.weight_at_spawning[a] * expZf_ABC);
221  }
222  sp_biomass /= api_females.kg_to_tonnes;
223 
224  // std::cout << "SB at F_ABC " << sp_biomass << "\tSBtarget " << SBtarget << "\tF_ABC " << F_ABC << "\tF_OFL " << F_OFL << "\tABC " << ABC << "\tOFL " << OFL << std::endl;
225 
226 
227  // adjust F_ABC and F_OFL
228  if (sp_biomass < SBfloor)
229  {
230  F_ABC = 0.0;
231  F_OFL = 0.0;
232  }
233  else if (sp_biomass < SBtarget)
234  {
235  F_ABC = FtargetABC * ((sp_biomass / SBtarget) - Tier3_alpha) / (static_cast<REAL_T> (1.0) - Tier3_alpha);
236  F_OFL = FtargetOFL * ((sp_biomass / SBtarget) - Tier3_alpha) / (static_cast<REAL_T> (1.0) - Tier3_alpha);
237  }
238  else
239  {
240  F_ABC = FtargetABC;
241  F_OFL = FtargetOFL;
242  }
243 
244 
245  // calculate ABC and OFL
246  ABC = 0.0;
247  OFL = 0.0;
248  for (int a = 0; a < Nages; ++a)
249  {
250  Zf_ABC = api_females.M[a] + (F_ABC * fsh_sel_f[a]);
251  expZf_ABC = atl::exp(static_cast<REAL_T> (-1.0) * Zf_ABC);
252  ABC += (api_females.N_proj[(season - 1) * Nages + a] * api_females.weight_at_catch_time[a] *
253  (static_cast<REAL_T> (1.0) - expZf_ABC) * (F_ABC * fsh_sel_f[a]) / Zf_ABC);
254 
255  Zm_ABC = api_males.M[a] + (F_ABC * fsh_sel_m[a]);
256  expZm_ABC = atl::exp(static_cast<REAL_T> (-1.0) * Zm_ABC);
257  ABC += (api_males.N_proj[(season - 1) * Nages + a] * api_males.weight_at_catch_time[a] *
258  (static_cast<REAL_T> (1.0) - expZm_ABC) * (F_ABC * fsh_sel_m[a]) / Zm_ABC);
259 
260 
261  Zf_OFL = api_females.M[a] + (F_OFL * fsh_sel_f[a]);
262  expZf_OFL = atl::exp(static_cast<REAL_T> (-1.0) * Zf_OFL);
263  OFL += (api_females.N_proj[(season - 1) * Nages + a] * api_females.weight_at_catch_time[a] *
264  (static_cast<REAL_T> (1.0) - expZf_OFL) * (F_OFL * fsh_sel_f[a]) / Zf_OFL);
265 
266  Zm_OFL = api_males.M[a] + (F_OFL * fsh_sel_m[a]);
267  expZm_OFL = atl::exp(static_cast<REAL_T> (-1.0) * Zm_OFL);
268  OFL += (api_males.N_proj[(season - 1) * Nages + a] * api_males.weight_at_catch_time[a] *
269  (static_cast<REAL_T> (1.0) - expZm_OFL) * (F_OFL * fsh_sel_m[a]) / Zm_OFL);
270  }
271  ABC /= api_females.kg_to_tonnes;
272  OFL /= api_females.kg_to_tonnes;
273  }
274 
275  }
276 
277  return std::make_tuple(F_ABC, ABC, F_OFL, OFL);
278  }

281  {
282  return "NPFMC Tier 3 HCR";
283  }
